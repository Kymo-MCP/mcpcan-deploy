Quick Installation

Add Helm Repository

# Add MCPCan Helm repository
helm repo add mcpcan https://kymo-mcp.github.io/mcpcan-deploy/

# Update repository
helm repo update

Install MCPCan

Option 1: Install from Helm Repository (Recommended)

# Install stable release from remote repository
helm install mcpcan mcpcan/mcpcan-deploy

# Custom installation with public IP
helm install mcpcan mcpcan/mcpcan-deploy \
  --set global.publicIP=192.168.1.100 \
  --set infrastructure.mysql.auth.rootPassword=secure-password \
  --set infrastructure.redis.auth.password=secure-password

# Custom installation with domain
helm install mcpcan mcpcan/mcpcan-deploy \
  --set global.domain=mcp.example.com \
  --set infrastructure.mysql.auth.rootPassword=secure-password \
  --set infrastructure.redis.auth.password=secure-password

Option 2: Install from Local Chart (Development)

# Clone repository first
git clone https://github.com/Kymo-MCP/mcpcan-deploy.git
cd mcpcan-deploy

# Install from local Chart directory
helm install mcpcan ./helm/

# Custom installation with local Chart
helm install mcpcan ./helm/ \
  --set global.publicIP=192.168.1.100 \
  --set infrastructure.mysql.auth.rootPassword=secure-password \
  --set infrastructure.redis.auth.password=secure-password

Verify Installation

# Check Pod status
kubectl get pods -l app.kubernetes.io/instance=mcpcan

# Check service status
kubectl get services -l app.kubernetes.io/instance=mcpcan
